// Add event listener to the "Add to Cart" buttons const addToCartButtons = document.querySelectorAll('.btn-primary'); addToCartButtons.forEach(button => { button.addEventListener('click', () => { const product = button.parentNode.parentNode; const productName = product.querySelector('.card-title').textContent; const productPrice = product.querySelector('.card-text').textContent; console.log(`Added ${productName} to cart for ${productPrice}`); }); });
